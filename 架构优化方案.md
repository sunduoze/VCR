# VCR 项目架构优化方案

## 一、问题背景

**当前性能瓶颈**：
- 大数据量（250K 点）时帧率仅 2 FPS，严重卡顿
- 数据接收与显示未分离，存在竞态条件
- 数据丢失率高

**优化目标**：
- 帧率提升至 30+ FPS
- 数据丢失率降低 50 倍
- 消除数据处理与渲染的耦合

---

## 二、架构优化方案（7 种）

### 方案 1：双缓冲隔离（推荐优先实施）⭐⭐⭐

#### 架构图

\\\
┌─────────────────┐     ┌─────────────────┐
│  Rust 接收线程  │     │  Flutter UI 线程 │
│  (独立线程)     │     │  (主 Isolate)    │
└────────┬────────┘     └────────┬────────┘
         │                       │
         ↓                       ↓
   [后台缓冲区 B]          [前台缓冲区 A]
   (write_buffer)          (read_buffer)
         │                       │
         └──────────┬────────────┘
                    │
                    ↓
              [定时交换] (16ms)
\\\

#### 实现要点

**Rust 侧**：
\\\ust
// 双缓冲结构
lazy_static! {
    static ref BUFFER_A: Mutex<Vec<PlotChannel>> = Mutex::new(Vec::new());
    static ref BUFFER_B: Mutex<Vec<PlotChannel>> = Mutex::new(Vec::new());
    static ref USE_BUFFER_A: AtomicBool = AtomicBool::new(true);
}

// 接收线程写入后台缓冲区
fn data_receiver_thread() {
    loop {
        let data = receive_data();  // 从串口/网络接收数据
        let mut buf = if USE_BUFFER_A.load(Ordering::SeqCst) {
            BUFFER_B.lock().unwrap()  // 写入后台 B
        } else {
            BUFFER_A.lock().unwrap()  // 写入后台 A
        };
        *buf = data;  // 更新后台缓冲区
    }
}

// Flutter 侧读取前台缓冲区
pub fn get_plot_data() -> Vec<PlotChannel> {
    let use_a = USE_BUFFER_A.load(Ordering::SeqCst);
    let buf = if use_a {
        BUFFER_A.lock().unwrap()  // 读取前台 A
    } else {
        BUFFER_B.lock().unwrap()  // 读取前台 B
    };
    buf.clone()  // 返回副本（或零拷贝）
}
\\\

**Flutter 侧**：
\\\dart
// 定时器回调（16ms）
void _fetchRealData() {
  final useA = RustLib.instance.api.getUseBufferA();  // 获取当前前台缓冲区标识
  final data = useA ? RustLib.instance.api.getBufferA() : RustLib.instance.api.getBufferB();
  
  setState(() {
    _channels = data;  // 更新 UI
  });
  
  // 交换缓冲区（下次读取另一个）
  RustLib.instance.api.swapBuffers();
}
\\\

#### 优点
- ✅ **数据处理和绘图完全隔离**，无锁竞争
- ✅ **实现相对简单**（Flutter 侧用 \List<Point>\ 交换即可）
- ✅ **适用于大数据量**（250K 点）

#### 缺点
- ❌ 需要定时交换缓冲区（可能引入延迟）
- ❌ 如果交换频率太低，绘图数据可能不连续

#### 适用场景
- 大数据量实时绘图，对延迟要求不高（>16ms）

// 独立 Isolate 入口
void dataProcessIsolate(SendPort sendPort) {
  final receiver = ReceivePort();
  sendPort.send(receiver.sendPort);
  
  receiver.listen((message) {
    // 接收原始数据
    final rawData = message['data'];
    // 数据处理（解析、滤波、降采样等）
    final processedData = processData(rawData);
    // 发送处理后的数据给 UI 线程
    sendPort.send(processedData);
  });
}

// 独立线程 + 双缓冲
lazy_static! {
    static ref BUFFER_A: Mutex<Vec<PlotChannel>> = Mutex::new(Vec::new());
    static ref BUFFER_B: Mutex<Vec<PlotChannel>> = Mutex::new(Vec::new());
    static ref USE_BUFFER_A: AtomicBool = AtomicBool::new(true);
}

// 独立线程：数据接收 + 处理
fn start_data_receiver() {
    thread::spawn(|| {
        loop {
            let data = receive_data();  // 从串口/网络接收数据
            let processed = process_data(data);  // 数据处理（解析、滤波、降采样）
            
            let mut buf = if USE_BUFFER_A.load(Ordering::SeqCst) {
                BUFFER_B.lock().unwrap()
            } else {
                BUFFER_A.lock().unwrap()
            };
            *buf = processed;
        }
    });
}

// FFI 接口：Flutter 侧读取数据
pub fn get_plot_data() -> Vec<PlotChannel> {
    let use_a = USE_BUFFER_A.load(Ordering::SeqCst);
    let buf = if use_a {
        BUFFER_A.lock().unwrap()
    } else {
        BUFFER_B.lock().unwrap()
    };
    buf.clone()
}

// GPU 渲染器：将数据上传到 GPU 纹理
fn render_waveform_to_texture(data: &[PlotChannel]) -> GpuTexture {
    // 1. 创建 GPU 缓冲区
    let vertex_buffer = device.create_buffer_init(&...);
    
    // 2. 渲染到纹理
    let mut encoder = device.create_command_encoder(&...);
    {
        let mut render_pass = encoder.begin_render_pass(&...);
        render_pass.set_pipeline(render_pipeline);
        render_pass.set_vertex_buffer(0, vertex_buffer.slice(...));
        render_pass.draw(0..data.len() as u32, 0..1);
    }
    
    // 3. 提交命令
    queue.submit(std::iter::once(encoder.finish()));
    
    // 4. 返回 GPU 纹理
    GpuTexture { ... }
}

void paint(Canvas canvas, Size size) {
  if (_gpuWaveformImage != null) {
    canvas.drawImage(_gpuWaveformImage!, Offset.zero, Paint());  // GPU 渲染

    // CPU 渲染（回退方案）
    for (var channel in _channels) {
      final path = _calculatePath(channel);
      canvas.drawPath(path, paint);
    }
  }
}

### 方案 7：环形缓冲区（Ring Buffer）⭐⭐⭐

#### 架构图

\\\
┌───────────────┐     ┌───────────────────┐
│  Rust 接收线程  │     │  Flutter UI 线程   │
│  (独立线程)     │     │  (主 Isolate)      │
└────────┬──────┘     └────────┬──────────┘
         │                       │
         ↓                       │
   [环形缓冲区]                  │
   (read_pos, write_pos)        │
         │                       │
         └──────────┬────────────┘
                    │
                    ↓
              [读取最新 N 个点]
\\\

#### 实现要点

**Rust 侧**：
\\\ust
use std::sync::atomic::{AtomicUsize, Ordering};

const BUFFER_SIZE: usize = 250_000;  // 250K 点

lazy_static! {
    static ref RING_BUFFER: Mutex<Vec<f32>> = Mutex::new(vec![0.0; BUFFER_SIZE]);
    static ref WRITE_POS: AtomicUsize = AtomicUsize::new(0);
    static ref READ_POS: AtomicUsize = AtomicUsize::new(0);
}

// 接收线程写入
fn data_receiver_thread() {
    loop {
        let data = receive_data();  // 从串口/网络接收数据
        let mut buf = RING_BUFFER.lock().unwrap();
        let mut write_pos = WRITE_POS.load(Ordering::SeqCst);
        
        // 写入环形缓冲区
        for value in data {
            buf[write_pos] = value;
            write_pos = (write_pos + 1) % BUFFER_SIZE;
        }
        
        WRITE_POS.store(write_pos, Ordering::SeqCst);
    }
}

// Flutter 侧读取最新 N 个点
pub fn get_latest_data(n: usize) -> Vec<f32> {
    let buf = RING_BUFFER.lock().unwrap();
    let write_pos = WRITE_POS.load(Ordering::SeqCst);
    
    let start_pos = if write_pos >= n { write_pos - n } else { BUFFER_SIZE - (n - write_pos) };
    
    let mut result = Vec::with_capacity(n);
    for i in 0..n {
        let pos = (start_pos + i) % BUFFER_SIZE;
        result.push(buf[pos]);
    }
    
    result
}
\\\

**Flutter 侧**：
\\\dart
// 定时器回调（16ms）

  // 读取最新 1000 个点
  final data = RustLib.instance.api.getLatestData(1000);

    _channels = data;  // 更新 UI
  });
}
\\\

#### 优点
- ✅ **内存效率高**（不需要分配/释放，复用固定大小缓冲区）
- ✅ **实现相对简单**（比双缓冲稍复杂，但比独立线程简单）

#### 缺点
- ❌ **需要手动管理读写位置**（容易出错）
- ❌ **不适合数据量变化大的场景**（缓冲区大小固定）

#### 适用场景
- 数据量稳定（固定大小），内存效率要求高

### 6.1 方案 1（双缓冲隔离）完整代码

**Rust 侧（\data_receiver.rs\）**：

\\\ust
use lazy_static::lazy_static;
use std::sync::Mutex;
use std::sync::atomic::{AtomicBool, Ordering};

lazy_static! {
    static ref BUFFER_A: Mutex<Vec<PlotChannel>> = Mutex::new(Vec::new());
    static ref BUFFER_B: Mutex<Vec<PlotChannel>> = Mutex::new(Vec::new());
    static ref USE_BUFFER_A: AtomicBool = AtomicBool::new(true);
}

#[tokio::main]
async fn start_data_receiver() {
    tokio::spawn(async move {
        loop {
            let data = receive_data().await;  // 从串口/网络接收数据
            
            let mut buf = if USE_BUFFER_A.load(Ordering::SeqCst) {
                BUFFER_B.lock().unwrap()
            } else {
                BUFFER_A.lock().unwrap()
            };
            
            *buf = data;  // 更新后台缓冲区
        }
    });
}

pub fn get_plot_data() -> Vec<PlotChannel> {
    let use_a = USE_BUFFER_A.load(Ordering::SeqCst);
    let buf = if use_a {
        BUFFER_A.lock().unwrap()
    } else {
        BUFFER_B.lock().unwrap()
    };
    buf.clone()
}

pub fn swap_buffers() {
    let current = USE_BUFFER_A.load(Ordering::SeqCst);
    USE_BUFFER_A.store(!current, Ordering::SeqCst);
}

pub fn gpu_render_waveform(data: &[PlotChannel]) -> Result<GpuTexture, String> {
    // 1. 创建 GPU 缓冲区
    let vertex_buffer = device.create_buffer_init(&wgpu::util::BufferInitDescriptor {
        label: Some("Waveform Vertex Buffer"),
        contents: bytemuck::cast_slice(data),
        usage: wgpu::BufferUsage::VERTEX | wgpu::BufferUsage::COPY_DST,
    });
    
    // 2. 渲染到纹理
    let mut encoder = device.create_command_encoder(&wgpu::CommandEncoderDescriptor {
        label: Some("Waveform Render Encoder"),
    });
    
    {
        let render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
            label: Some("Waveform Render Pass"),
            color_attachments: &[wgpu::RenderPassColorAttachment {
                view: &texture_view,
                resolve_target: None,
                ops: wgpu::Operations {
                    load: wgpu::LoadOp::Clear(wgpu::Color::BLACK),
                    store: true,
                },
            }],
            depth_stencil_attachment: None,
        });
        
        render_pass.set_pipeline(render_pipeline);
        render_pass.set_vertex_buffer(0, vertex_buffer.slice(..));
        render_pass.draw(0..data.len() as u32, 0..1);
    }
    
    // 3. 提交命令
    queue.submit(std::iter::once(encoder.finish()));
    
    // 4. 返回 GPU 纹理
    Ok(GpuTexture { texture, texture_view })
}
